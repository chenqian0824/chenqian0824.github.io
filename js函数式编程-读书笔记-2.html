闭包的抽象
```
function plucker (FIELD) {
    return function (obj) {
        return (obj && obj[FIELD])
    }
}
let best = {title: 'infinite jest', author: 'dfw'}
let getTitle = plucker('title')
let dd = getTitle(best);
console.log(dd) // infinite jest
```
